consider everything that grows

Note

Most of the headings are taken from the verses of Shakespeare’s Sonnet 15

Population pyramids give us a clear picture of the growth and “health” of a country, its states down to its counties. I came across the pyramids in one of my geography classes while living in Cuba. I thought they were pretty cool because I was obsessed with history1, thought they were connected to Ancient Egypt, and I could actually see how human movement into and out of existence affected my society and the economy in a very real sense. If the person who sold rice, sugar, flour, meat, or eggs, passed away, well now you have to go without it or find someone else who, taking advantage of the new vacancy, will jack up the prices. My generation saw a considerable decline in the number of children. We just simply didn’t have the means to feed and keep multiple kids. Senior citizens, likewise, weren’t many because we lacked the resources to effectively deal with naturally-occurring diseases and ailments in the elderly population. Most of my grandparents lived to be 90+, but they were the exception. I could clearly see all of this in the population pyramid of my generation.

1 especially Ancient Egypt, the Bible, and the Roman Empire.

cu_path <- here::here("blog/2024/07/pyramids/data/cuba_95.txt")
cuba_95 <- read.delim(cu_path, sep = "")|>
  pivot_longer(cols = c(M,F),
                                      names_to = "gender",
                                      values_to = "population") |>
  mutate(gender = case_when(gender == "M" ~ "Male",
                            gender == "F" ~ "Female"),
         total = sum(population),
         percent = population/total,
         Age = fct_inorder(Age)) |>
  mutate(percent = if_else(gender == "Female", percent, -percent)) |>
  group_by(gender)

cu_male95 <- cuba_95 |> filter(gender == "Male") |>
  ggplot(aes(percent,Age)) +
  geom_col(fill = va_clrs[1]) +
  annotate(geom = "label",
           family = "Franklin Gothic Medium",
           x = -0.05,
           y = 17,
           label = "Men",
           fill = va_clrs[3],
           color = "white",
           label.size = 0,
           label.padding = unit(0.3,"lines")) +
  scale_x_continuous(
    labels = function(x) label_percent(accuracy = 1)(abs(x)),
    limits = c(-0.06,0)
  )

cu_female95 <- cuba_95 |> filter(gender == "Female") |>
  ggplot(aes(percent,Age)) +
  geom_col(fill = va_clrs[3]) +
  annotate(geom = "label",
           family = "Franklin Gothic Medium",
           x = 0.05,
           y = 17,
           label = "Women",
           fill = va_clrs[5],
           color = "gray30",
           label.size = 0,
           label.padding = unit(0.3, "lines")) +
  scale_x_continuous(
    labels = function(x) label_percent(accuracy = 1)(abs(x)),
    limits = c(0,0.06)
  )

age_labels <- tibble( age = c( "0-4","5-9","10-14",
      "15-19", "20-24","25-29",
      "30-34","35-39","40-44",
      "45-49", "50-54", "55-59",
      "60-64","65-69", "70-74",
      "75-79","80-84", "85-89",
      "90+" )) |>
  mutate(age = fct_inorder(age)) 

age_plot <- age_labels |>
  ggplot(aes(x = 1,y = age, label = age)) +
  geom_text(family = "Franklin Gothic Medium", size = 4.5) +
  theme_void()

cu_male95 +
  age_plot +
cu_female95 +
  plot_layout(widths = c(7.5,1,7.5))
